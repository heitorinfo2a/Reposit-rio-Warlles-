Algoritmo MediaGasolinaMensal

Início
    // Declaração de variáveis
    Real: m1, m2, m3, media_aritmetica

    // Entrada de dados
    Escreva("Informe o gasto de gasolina do 1º mês: ")
    Leia(m1)

    Escreva("Informe o gasto de gasolina do 2º mês: ")
    Leia(m2)

    Escreva("Informe o gasto de gasolina do 3º mês: ")
    Leia(m3)

    // Processamento
    media_aritmetica ← (m1 + m2 + m3) / 3

    // Saída de dados
    Escreva("A média é ", media_aritmetica)

Fim
