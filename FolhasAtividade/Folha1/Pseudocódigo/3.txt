Algoritmo MediaAritmetica

Início
    // Declaração de variáveis
    Real: n1, n2, n3, n4, media_aritmetica

    // Entrada de dados
    Escreva("Informe a 1ª nota: ")
    Leia(n1)

    Escreva("Informe a 2ª nota: ")
    Leia(n2)

    Escreva("Informe a 3ª nota: ")
    Leia(n3)

    Escreva("Informe a 4ª nota: ")
    Leia(n4)

    // Processamento
    media_aritmetica ← (n1 + n2 + n3 + n4) / 4

    // Saída de dados
    Escreva("A média é ", media_aritmetica)

Fim
