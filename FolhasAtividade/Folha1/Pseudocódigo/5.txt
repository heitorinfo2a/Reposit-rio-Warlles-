Algoritmo PesoIdeal

Início
    // Declaração de variáveis
    Real: altura, peso
    Constante Real: k ← 72.7
    Constante Real: B ← 58

    // Entrada de dados
    Escreva("Informe a altura: ")
    Leia(altura)

    // Processamento
    peso ← k * altura - B

    // Saída de dados
    Escreva("O seu peso ideal é ", peso)

Fim
