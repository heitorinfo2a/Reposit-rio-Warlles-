Algoritmo TrocarValores

Início
    // Declaração de variáveis
    Real: x, y, trocar

    // Entrada de dados
    Escreva("Insira o valor de x: ")
    Leia(x)

    Escreva("Insira o valor de y: ")
    Leia(y)

    // Exibição antes da troca
    Escreva("ANTES DA TROCA")
    Escreva(x, " ", y)

    // Processamento da troca
    trocar ← x
    x ← y
    y ← trocar

    // Exibição após a troca
    Escreva("DEPOIS DA TROCA")
    Escreva(x, " ", y)

Fim
